    const legend = svg
    .append("g")
    .attr("class", "legendLinear")
    .attr("transform", "translate(10,220)")

    const legendLinear = d3.legendColor()
    .shapeWidth(10)
    .shapeHeight(10)
    .shapePadding(5)
    .orient('vertical')
    .scale(scale_color)
    .title("Number of SDA dwellings")

    legend
    .select(".legendLinear")
    .call(legendLinear)
